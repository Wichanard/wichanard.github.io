<script>
  var app = document.getElementById('typewriter');

  var typewriter = new Typewriter(app, {
    loop: true,
    delay: 75,
  });

  typewriter
    .pauseFor(500)
    .typeString('สวัสดีครับ, ผมชื่อ **[WICHANARD KUMSANG]**')
    .pauseFor(1000)
    .deleteAll()
    .typeString('ผมสนใจเรื่อง **Artificial Intelligence**')
    .pauseFor(1000)
    .deleteAll()
    .typeString('ยินดีที่ได้รู้จักครับ!')
    .pauseFor(2500)
    .start();
</script>
